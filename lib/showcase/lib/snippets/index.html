1 — Numeric pipeline

Filter evens, square them, sum → 120

nums = [1, 2, 3, 4, 5, 6, 7, 8]

result = sum(x**2 for x in nums if x % 2 == 0)

print(result)  # 120

    result := 0

            result += x * x
        }
    }

    fmt.Println(result) // 120

    f"{name} is {age}"
    for name, age in people
    if age >= 18

package main

import "fmt"

type Person struct {
    Name string
    Age  int
}

func main() {
    people := []Person{
        {"Alice", 30}, {"Bob", 15}, {"Carol", 22},
    }

    var adults []string
    for _, p := range people {
        if p.Age >= 18 {
            adults = append(adults,
                fmt.Sprintf("%s is %d", p.Name, p.Age))
        }
    }

    fmt.Println(adults)
    // [Alice is 30 Carol is 22]
}
